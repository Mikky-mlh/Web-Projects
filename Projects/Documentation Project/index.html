# Optimized win detection using direction vectors def winning_move(board, piece): # Check all possible winning locations # Directions: horizontal, vertical, positive diagonal, negative diagonal directions = [(0, 1), (1, 0), (1, 1), (1, -1)] for r_start in range(ROW_COUNT): for c_start in range(COLUMN_COUNT): for dr, dc in directions: # Check if 4 consecutive pieces exist if 0 <= r_start + 3*dr < ROW_COUNT and \ 0 <= c_start + 3*dc < COLUMN_COUNT: if all(board[r_start + i*dr][c_start + i*dc] == piece for i in range(4)): return True return False
